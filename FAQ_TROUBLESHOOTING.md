function convertToCSVWithBOM(data) {
  const csv = data.map(row => {
    return row.map(cell => {
      let value = cell !== null && cell !== undefined ? cell.toString() : '';
      value = value.replace(/"/g, '""');
      
      if (value.includes(',') || value.includes('\n') || value.includes('"')) {

      }
      
      return value;
